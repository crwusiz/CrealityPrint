add_custom_target(packaging COMMENT "Packaging Projects...")
if(TARGET ${PROJECT_DLL})
    add_dependencies(packaging ${PROJECT_DLL})
elseif(TARGET ${PROCESS_NAME})
    add_dependencies(packaging ${PROCESS_NAME})
else()
    message(WARNING "Packaging target: no executable target named ${PROJECT_DLL} or ${PROCESS_NAME}; packaging will run without build dependency.")
endif()
IF(APPLE)
    include(macx.cmake)
elseif (WIN32)
    message(STATUS "Windows Packaging")
    if(NOT DEFINED PACKAGE_CONF)
        set(PACKAGE_CONF package.cmake)
    endif()
    include(${PACKAGE_CONF})
    if (EXISTS "${CMAKE_CURRENT_LIST_DIR}/squirrel.cmake")
        include(${CMAKE_CURRENT_LIST_DIR}/squirrel.cmake)
    endif()
endif()
